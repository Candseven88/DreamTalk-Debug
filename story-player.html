// 播放权限限制系统 function speakStory() { // 获取用户订阅等级 const userPlan = localStorage.getItem('userPlan') || 'free'; // 如果是付费用户，直接允许播放 if (userPlan !== 'free' && ['starter', 'plus', 'pro'].includes(userPlan.toLowerCase())) { playAudioContent(); // 调用原有的播放函数 return; } // 对免费用户进行播放次数限制 const today = new Date().toISOString().split('T')[0]; // 获取当前日期，格式：YYYY-MM-DD // 获取并解析每日播放记录 let dailyPlayLog = {}; try { dailyPlayLog = JSON.parse(localStorage.getItem('dailyPlayLog') || '{}'); } catch (e) { dailyPlayLog = {}; } // 获取今日播放次数 const todayPlayCount = dailyPlayLog[today] || 0; // 检查是否超出限制 if (todayPlayCount >= 5) { // 显示升级提示 alert("Free users can only play 5 stories per day. Please upgrade your plan for unlimited access."); return; } // 允许播放，并更新播放记录 playAudioContent(); // 调用原有的播放函数 // 更新播放记录 dailyPlayLog[today] = todayPlayCount + 1; localStorage.setItem('dailyPlayLog', JSON.stringify(dailyPlayLog)); } // 假设这是原有的播放函数，根据实际情况替换 function playAudioContent() { // 原有的播放逻辑 // 如果页面中已有播放函数，请将此函数替换为调用原有函数 console.log("Playing audio content..."); // 如果页面使用的是其他播放函数名称，请相应修改 speakStory() 中的调用 } // 将原有播放按钮的点击事件修改为调用 speakStory document.addEventListener('DOMContentLoaded', function() { const playButton = document.getElementById('play-button'); // 根据实际情况修改选择器 if (playButton) { // 移除原有事件监听器（如果有） const newPlayButton = playButton.cloneNode(true); playButton.parentNode.replaceChild(newPlayButton, playButton); // 添加新的事件监听器 newPlayButton.addEventListener('click', speakStory); } });
